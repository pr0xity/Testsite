Webserver

This is the first hing I have written in GOlang. It was the way I chose to
present my first website.

package main

import (
"fmt"
"log"
"net/http"

"github.com/gorilla/mux"
)

func main() {
r := mux.NewRouter()

r.PathPrefix("/").Handler(http.FileServer(http.Dir("./public/")))
http.Handle("/", r)

// Starting server
fmt.Printf("Starting server on port \"4000\" ...")
log.Fatal(http.ListenAndServe(":4000", nil))
}
